on_actions = {
  on_annex = {
    effect = {
      recheck_annexations = yes
    }
  }

  on_peaceconference_ended = {
    effect = {
      recheck_annexations = yes
    }
  }

  on_state_control_changed = {
    effect = {
      if = {
        limit = {
          owns_state = FROM.FROM
          is_in_array = { global.annexation_relevant_states = FROM.FROM }
        }
        recheck_annexations = yes
      }
    }
  }
  on_daily = {
    effect = {
      if = {
        limit = { check_variable = { var:coalition_partners > 0 } }
        set_coalition_government_popularity = yes
      }
    }
  }
  on_startup = {
    effect = {
      every_possible_country = {
				#Party Popularity Setup
				if = {
					limit = { has_variable = political_power_ideology }
					map_party_popularity_PP_gain_to_ideology = yes
				}
				else = {
					default_party_popularity_PP_gain = yes
				}
        #Remove Party Pop
        remove_ideas = {
          party_pop_0_00
          party_pop_0_10
          party_pop_0_20
          party_pop_0_30
          party_pop_0_40
          party_pop_0_50
          party_pop_0_60
          party_pop_0_70
          party_pop_0_80
          party_pop_0_90
        }
      }
    }
  }
}
